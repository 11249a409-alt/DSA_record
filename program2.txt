Aim: To write a C program to implement binary search.

Algorithm:
1. Start.
2. Read the number of elements in a sorted array.
3. Read the elements of the array.
4. Read the element to be searched (key).
5. Set 'first' to 0 and 'last' to the last index.
6. While 'first' is less than or equal to 'last', repeat the following steps.
7. Calculate 'middle' as the average of 'first' and 'last'.
8. If the key matches the element at 'middle', print the location and exit.
9. If the key is less than the element at 'middle', update 'last' to 'middle - 1'.
10. If the key is greater than the element at 'middle', update 'first' to 'middle + 1'.
11. If the element is not found, print a message.
12. Stop.

#include <stdio.h>

int main() {
    int arr[100], n, i, key, first, last, middle;

    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    printf("Enter the elements of the array in sorted order:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter the element to search: ");
    scanf("%d", &key);

    first = 0;
    last = n - 1;
    middle = (first + last) / 2;

    while (first <= last) {
        if (arr[middle] < key) {
            first = middle + 1;
        } else if (arr[middle] == key) {
            printf("Element found at index %d\n", middle);
            break;
        } else {
            last = middle - 1;
        }
        middle = (first + last) / 2;
    }

    if (first > last) {
        printf("Element not found\n");
    }

    return 0;
}

Output:
